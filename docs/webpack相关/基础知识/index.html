webpack

webpack 是一个前端打包工具，主要作用是把多个 js 文件合并成一个 js 文件，减少 http 请求，提高页面加载速度。

1. 安装​

npm install webpack webpack-cli -D

2. 配置​

// webpack.config.js

const path = require('path') //nodejs模块 处理路径的

// 基本框架
module.exports = {
  entry: '', //入口文件
  output: {}, //出口文件
  module: {}, //loader 配置
  plugins: [], //插件配置
  devServer: {} //开发服务器配置
  mode: 'development' //开发模式
  devtool: 'source-map'//源代码映射
}

1.1. entry 入口​

入口起点(entry point) 指示 webpack 应该使用哪个模块，来作为构建其内部 依赖图(dependency graph) 的开始。进入入口起点后，webpack 会找出有哪些模块和库是入口起点（直接和间接）依赖的。

默认值是 ./src/index.js，但你可以通过在 webpack configuration 中配置 entry 属性，来指定一个（或多个）不同的入口起点。例如：

webpack.config.js​

单入口文件配置

const path = require('path') //nodejs模块 处理路径的
//采取相对路径
module.exports = {
  entry: './path/to/my/entry/file.js'
}

// 利用path模块
/**
 * 配置模块导出
 *
 * 该模块导出了一个对象，包含了项目的入口配置信息。
 * 其中，entry属性指定了项目的入口文件路径，这是Webpack开始构建的起点。
 */
module.exports = {
  entry: path.resolve(__dirname, 'src/index.js')
}

// 对象形式配置
module.exports = {
  entry: {
    main: './path/to/my/entry/file.js'
  }
}

多入口文件配置

const path = require('path')
const entryList = ['./src/file_1.js', './src/file_2.js'].map(item => path.resolve(__dirname, item))
module.exports = {
  entry: entryList,
  output: {
    filename: 'bundle.js'
  }
}

// 对象语法 用法：entry: { <entryChunkName> string | [string] } | {}
module.exports = {
  entry: {
    app: './src/app.js',
    adminApp: './src/adminApp.js'
  }
}

具体代码

// 路径模块 专门处理路径的
const path = require('path')
// 操作系统模块
const os = require('os')
// terser 压缩js代码
const TerserPlugin = require('terser-webpack-plugin')
// 引入eslint插件
const ESLintPlugin = require('eslint-webpack-plugin')
// const { Generator } = require('webpack')

// html插件 作用：自动生成html文件，并引入打包后的js文件
const HtmlWebpackPlugin = require('html-webpack-plugin')

// 压缩css 插件 作用 作用：压缩css
const MiniCssExtractPlugin = require('mini-css-extract-plugin')

// 提取css 插件 作用：提取css代码
const CssMinimizeCssPlugin = require('css-minimizer-webpack-plugin') // 压缩css

// 压缩图片插件
const ImageMinimizerPlugin = require('image-minimizer-webpack-plugin')

const threads = os.cpus().length // 获取cpu核数
function getSyleLoader(pre) {
  return [
    MiniCssExtractPlugin.loader,
    'css-loader',
    // css兼容低版本浏览器
    {
      loader: 'postcss-loader',
      options: {
        postcssOptions: {
          plugins: ['postcss-preset-env']
        }
      }
    },
    ...(pre && [pre])
  ]
}

module.exports = {
  // 入口 相对
  entry: './src/main.js',
  // 输出
  output: {
    // 输出路径 绝对路径
    // __dirname 当前文件的绝对路径 当前文件夹的绝对路径
    path: path.resolve(__dirname, '../dist'),
    // path: undefined,
    // 输出文件名
    // 入口文件名 打出输出文件名
    filename: 'js/bundle.js',
    // 输出文件名 打包后的文件名
    chunkFilename: 'static/js/[name].chunk.js',
    // 图片
    assetModuleFilename: 'static/images/[hash:10][ext][query]', // 输出图片名称
    // 删除dist文件夹
    clean: true
  },
  // 加载器
  module: {
    rules: [
      {
        // 匹配文件 匹配后缀名
        oneOf: [
          {
            test: /\.css$/, // 匹配文件 xxx.css
            // loader 顺序从右向左执行
            // style-loader 通过创建style标签将样式放入html中
            // css-loader 将css资源编译成commonjs模块到js中
            // 1.loader的配置
            use: getSyleLoader()
          },
          //   less-loader 将less文件编译成css文件
          {
            test: /\.les s$/i,
            use: getSyleLoader('less')
          },
          //   sass-loader 将sass文件编译成css文件
          {
            test: /\.s[ac]ss$/i,
            use: getSyleLoader('sass')
          },
          //  图片
          {
            test: /\.(png|jpg|gif)$/i,
            type: 'asset',
            parser: {
              // 配置图片大小
              dataUrlCondition: {
                // 小于8kb的图片转base64
                // 优点: 减少请求数量 缺点: 大文件会变base64
                maxSize: 8 * 1024 // 8kb
              }
            },
            generator: {
              filename: 'static/images/[name].[hash:10][ext]'
            }
          },
          // 字体 图标
          {
            test: /\.(ttf|woff2?|mp4|mp3|avi)$/,
            type: 'asset/resource'
            // generator: {
            //   filename: 'static/media/[name].[hash:10][ext]'
            // }
          },
          // babel 兼容性处理
          {
            test: /\.js$/,
            exclude: /node_modules/,
            use: [
              // 线程池 优化loader执行
              {
                loader: 'thread-loader',
                options: {
                  workers: threads // 进程数量
                }
              },
              {
                loader: 'babel-loader',
                options: {
                  presets: [
                    '@babel/preset-env',
                    {
                      useBuiltIns: 'usage', // 按需加载
                      corejs: 3
                    }
                  ],

                  plugins: ['@babel/plugin-transform-runtime'] // 减少代码体积
                }
              }
            ]
          }
        ]
      }
    ]
  },
  // 插件
  plugins: [
    new ESLintPlugin({
      // 检测规则 context 检测的目录
      context: path.resolve(__dirname, '../src'),
      // 检测排除项
      exclude: 'node_modules',
      cache: true, // 开启缓存
      // 缓存路径
      cacheLocation: path.resolve(__dirname, 'node_modules/.cache/.eslintcache')
    }),
    new HtmlWebpackPlugin({
      // 以源文件创建 html文件
      // 模板文件 自动引入打包输出的所有资源
      template: path.resolve(__dirname, '../public/index.html')
    }),
    // 提取css 样式 样式文件
    new MiniCssExtractPlugin({
      filename: 'static/css/main.css',
      chunkFilename: 'static/css/[name].chunk.css'
    }),
    // 图片压缩
    new ImageMinimizerPlugin({
      // 压缩图片
      minimizer: {
        implementation: ImageMinimizerPlugin.imageminGenerate,
        options: {
          plugins: [
            // gif压缩
            ['gifsicle', { interlaced: true }],
            // jpg压缩
            ['jpegtran', { progressive: true }],
            // png压缩
            ['optipng', { optimizationLevel: 5 }],
            // svg压缩
            ['svgo', { name: 'preset-default' }]
            // ['webp', { quality: 75 }]
          ]
        }
      }
    })
  ],
  optimiza: {
    // 压缩css
    minimizer: [
      // 压缩css
      new CssMinimizeCssPlugin(),
      // 压缩js
      new TerserPlugin({
        parallel: threads // 线程池
      })
    ]
  },
  optimization: {
    splitChunks: {
      chunks: 'all', //对所有模式进行分割
      cacheGroups: {
        // 缓存组
        commons: {
          name: 'commons', // 抽离出来的文件名
          chunks: 'initial', // 只对入口文件进行分割
          minSize: 0, // 分割最小字节
          minChunks: 2 // 最少被引用次数
        }
      }
    },
    // 运行 chunks
    runtimeChunk: {
      name: entrypoint => `runtime~${entrypoint.name}.js`
    }
  },

  // 模式
  mode: 'development',
  // stats
  stats: {
    children: true
  }
}
